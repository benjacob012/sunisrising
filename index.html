function todayStr(){return new Date().toISOString().split('T')[0];}

function daysDiff(a,b){return Math.ceil((new Date(b+'T00:00:00')-new Date(a+'T00:00:00'))/86400000);}

function isUrgent(p){return p.status==='active'&&daysDiff(todayStr(),p.deadline)<=3;}

function isOverdue(p){return p.status==='active'&&daysDiff(todayStr(),p.deadline)<0;}

function isLateStart(p){return p.status==='active'&&daysDiff(p.startDate,todayStr())>=2;}

function br_renderCalendar(){

  const grid=document.getElementById('br-calendar-grid');if(!grid)return;grid.innerHTML='';

  const year=br_calViewDate.getFullYear(),month=br_calViewDate.getMonth();

  const title=document.getElementById('br-cal-title');

  if(title)title.innerText=new Intl.DateTimeFormat('id-ID',{month:'long'}).format(br_calViewDate).toUpperCase()+' '+year;

  const daysInMonth=new Date(year,month+1,0).getDate();let firstDow=new Date(year,month,1).getDay();firstDow=firstDow===0?6:firstDow-1;

  const today=todayStr();

  for(let i=0;i<firstDow;i++)grid.appendChild(document.createElement('div'));

  for(let day=1;day<=daysInMonth;day++){

    const dk=`${year}-${String(month+1).padStart(2,'0')}-${String(day).padStart(2,'0')}`;

    const reaches=allBrandsData.filter(b=>b.date===dk);

    const follows=allBrandsData.filter(b=>b.follow===dk);

    const overdue=follows.filter(b=>dk<today&&!['closed','rejected','ghost'].includes(b.status));

    const cell=document.createElement('div');cell.className='br-cal-day';

    if(dk===today)cell.classList.add('br-today');

    if(overdue.length>0)cell.classList.add('br-has-overdue');

    else if(follows.length>0)cell.classList.add('br-has-follow');

    else if(reaches.length>0)cell.classList.add('br-has-reach');

    const num=document.createElement('span');num.textContent=day;cell.appendChild(num);

    const all=[...new Map([...reaches,...follows].map(b=>[b.id,b])).values()];

    if(all.length>0){

      const dr=document.createElement('div');dr.className='br-dot-row';

      all.slice(0,5).forEach(b=>{const dot=document.createElement('span');dot.className='br-dot';const isOD=b.follow===dk&&dk<today&&!['closed','rejected','ghost'].includes(b.status);dot.style.background=isOD?'var(--red)':b.follow===dk?'var(--amber)':'var(--teal)';dr.appendChild(dot);});

      cell.appendChild(dr);

      const tip=document.createElement('div');tip.className='br-cal-tooltip';

      tip.textContent=[...reaches.map(b=>'📤 '+b.name),...follows.map(b=>'🔔 '+b.name)].join(' · ');cell.appendChild(tip);

    }

    cell.addEventListener('click',()=>{const names=[...reaches.map(b=>'Reach: '+b.name),...follows.map(b=>'Follow: '+b.name)].join(' | ');if(names)showToast(names.length>50?names.substring(0,47)+'...':names);});

    grid.appendChild(cell);

  }

}
